  <script>
    // 1) Navigation between views
    const linkGames = document.getElementById('link-games');
    const linkProxy = document.getElementById('link-proxy');
    const viewGames = document.getElementById('view-games');
    const viewProxy = document.getElementById('view-proxy');

    function showGames() {
      linkGames.classList.add('active');
      linkProxy.classList.remove('active');
      viewGames.style.display = '';
      viewProxy.style.display = 'none';
    }
    function showProxy() {
      linkProxy.classList.add('active');
      linkGames.classList.remove('active');
      viewProxy.style.display = '';
      viewGames.style.display = 'none';
    }

    linkGames.addEventListener('click', e => { e.preventDefault(); showGames(); });
    linkProxy.addEventListener('click', e => { e.preventDefault(); showProxy(); });

    // 2) Populate 80 game entries
    const games = [];
    for (let i = 1; i <= 80; i++) {
      games.push({
        name: `Unblocked Game ${i}`,
        // Replace the URL below with a real unblocked game URL or proxy URL as needed
        url: `https://example.com/unblocked-game-${i}`
      });
    }
    const list = document.getElementById('games-list');
    games.forEach(g => {
      const div = document.createElement('div');
      div.className = 'game-item';
      div.innerHTML = `
        <a href="#" onclick="loadGame('${g.url}');return false;">${g.name}</a>
        <iframe class="game-iframe" src="${g.url}" allowfullscreen></iframe>
      `;
      list.appendChild(div);
    });

    function loadGame(url) {
      // find the first iframe and load the selected game
      const iframe = document.querySelector('.game-iframe');
      iframe.src = url;
    }

    // 3) Proxy search function
    function searchProxy(e) {
      e.preventDefault();
      const q = encodeURIComponent(document.getElementById('proxy-query').value);
      // DuckDuckGo search URL
      const url = `https://duckduckgo.com/?q=${q}`;
      // Open in new tab so it's unblocked
      window.open(url, '_blank');
      // Optionally also load in the embedded iframe
      document.getElementById('proxy-frame').src = url;
    }

    // On load, check hash
    if (location.hash === '#proxy') showProxy();
  </script>
